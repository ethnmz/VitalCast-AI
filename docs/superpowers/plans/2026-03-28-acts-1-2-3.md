/**
 * Returns top 3 risk factors sorted by severity, with coaching copy.
 * @param {object} metrics — output of computeMetrics()
 * @returns {Array<{id, label, value, unit, level, description, tip}>}
 */
export function getRiskFactors(metrics) {
  const factors = []

  // CVD Risk
  const cvdLevel = metrics.cvdRisk >= 20 ? 'high' : metrics.cvdRisk >= 10 ? 'moderate' : 'low'
  factors.push({
    id: 'cvd',
    label: 'Cardiovascular Risk',
    value: metrics.cvdRisk,
    unit: '% 10-yr',
    level: cvdLevel,
    score: metrics.cvdRisk,
    description: cvdLevel === 'high'
      ? 'Your Framingham score is above the 20% threshold — clinical attention recommended.'
      : cvdLevel === 'moderate'
      ? 'Moderate risk. Lifestyle changes can move this significantly within 90 days.'
      : 'Good cardiovascular health. Keep up the activity and sleep consistency.',
    tip: 'Reduce resting HR by 5 bpm → drops risk by ~1.2%',
  })

  // HRV / Stress
  const stressLevel = metrics.stressIndex >= 70 ? 'high' : metrics.stressIndex >= 45 ? 'moderate' : 'low'
  factors.push({
    id: 'stress',
    label: 'HRV Stress Index',
    value: metrics.stressIndex,
    unit: '/ 100',
    level: stressLevel,
    score: metrics.stressIndex,
    description: stressLevel === 'high'
      ? 'Your autonomic nervous system is under significant strain — recovery is impaired.'
      : stressLevel === 'moderate'
      ? 'Moderate stress load. Sleep quality and breathing exercises can help.'
      : 'Good HRV. Your nervous system is recovering well.',
    tip: '5 min box-breathing daily → +6 HRV pts in 30 days',
  })

  // Sleep
  const sleepLevel = metrics.psqi >= 10 ? 'high' : metrics.psqi >= 6 ? 'moderate' : 'low'
  factors.push({
    id: 'sleep',
    label: 'Sleep Quality',
    value: metrics.psqi,
    unit: 'PSQI',
    level: sleepLevel,
    score: sleepLevel === 'high' ? metrics.psqi * 5 : sleepLevel === 'moderate' ? metrics.psqi * 3 : metrics.psqi,
    description: sleepLevel === 'high'
      ? 'Poor sleep (PSQI > 10) compounds every other risk factor. This is priority #1.'
      : sleepLevel === 'moderate'
      ? 'Fair sleep. Consistent sleep timing is the single highest-leverage intervention.'
      : 'Good sleep. Protect this — it\'s doing heavy lifting for your recovery.',
    tip: 'Consistent bedtime ±15 min → PSQI improves ~1.5 pts/week',
  })

  // Sort highest severity first
  const order = { high: 0, moderate: 1, low: 2 }
  return factors.sort((a, b) => order[a.level] - order[b.level])
}

/**
 * Re-runs computeMetrics with overridden lifestyle vars to produce a "what-if" scenario.
 * @param {object} baseMetrics — current metrics from context
 * @param {object} overrides   — { isSmoker?, activityLevel?, avgSleepHours?, restingHR? }
 * @returns {object} new metrics object
 */
export function simulateChange(baseMetrics, overrides) {
  return computeMetrics({
    restingHR:      overrides.restingHR      ?? baseMetrics.restingHR,
    hrv:            baseMetrics.hrvScore,     // keep HRV stable
    avgSleepHours:  overrides.avgSleepHours  ?? baseMetrics.avgSleepHours,
    avgSteps:       baseMetrics.avgSteps,
    age:            baseMetrics.age,
    activityLevel:  overrides.activityLevel  ?? baseMetrics.activityLevel,
    isSmoker:       overrides.isSmoker       ?? false,
    dataSource:     baseMetrics.dataSource,
    parsedDays:     baseMetrics.parsedDays,
  })
}

/**
 * Returns a single weekly goal based on the top risk factor.
 * @param {Array} riskFactors — output of getRiskFactors()
 * @returns {{ goal, why, badge, badgeEmoji }}
 */
export function generateWeeklyPlan(riskFactors) {
  const top = riskFactors[0]
  const plans = {
    cvd: {
      goal: '30 min cardio, 4 days this week',
      why: 'Reduces your CVD risk by ~2% per month of consistency.',
      badge: 'Heart Starter',
      badgeEmoji: '❤️',
    },
    stress: {
      goal: '5 min box-breathing before bed, 5 nights',
      why: 'Shown to improve HRV by ~6–8 pts over 30 days.',
      badge: 'Calm Seeker',
      badgeEmoji: '🧘',
    },
    sleep: {
      goal: 'Consistent 10:30pm bedtime, 5 nights',
      why: 'Sleep timing consistency improves PSQI score ~1.5 pts/week.',
      badge: 'Sleep Builder',
      badgeEmoji: '🌙',
    },
  }
  return plans[top?.id] || plans.cvd
}
